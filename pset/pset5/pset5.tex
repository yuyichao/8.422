  \delta E=&-\sum_{i_a, i_b}\frac{1}{\hbar\omega^a_{ig}+\hbar\omega^b_{ig}}\abs{\frac{2e^2x^a_{ig}x^b_{ig}}{R^3}}^2\\
  =&-\frac{4e^4}{\hbar R^6}\sum_{i_a, i_b}\frac{\abs{x^a_{ig}}^2\abs{x^b_{ig}}^2}{\omega^a_{ig}+\omega^b_{ig}}
